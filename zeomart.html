function increment(id){ let foodDoo = JSON.parse(localStorage.getItem("fooditems")); foodDoo.forEach((list)=>{ if (list.id == id) { list.qty++ ; // list.total = (list.price * list.qty) let actualPrice = foodDoo.find(el=>el.name==list.name).price list.price = Number(list.price) + Number(actualPrice) } }) localStorage.setItem('fooditems', JSON.stringify(foodDoo)) myTableFood() } function decrement(id){ let foodDoo = JSON.parse(localStorage.getItem("fooditems")); foodDoo.forEach((list)=>{ if (list.id == id && list.qty >1 ) { list.qty--; let actualPrice = foodDoo.find(el=>el.name==list.name).price list.price = Number(list.price) - Number(actualPrice) } }) localStorage.setItem('fooditems', JSON.stringify(foodDoo)) myTableFood() }
